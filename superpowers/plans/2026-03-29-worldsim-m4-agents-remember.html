WorldSim M4: Agents Remember -- Implementation Plan ​

For agentic workers: REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.

Goal: Give agents a bounded memory system so they remember good resource locations, revisit them, and make smarter decisions. When no food is visible, agents use memory to navigate to remembered food locations instead of wandering randomly.

Architecture: New internal/ai/memory.go with a Memory struct (bounded ring buffer of entries). Each entry stores type, position, tick, and resource amount. Perception auto-records memories. Planner consults memory when vision fails. Memory entries are evicted oldest-first when capacity is reached.

Tech Stack: Pure Go, no new dependencies.

func TestMemory_BoundedCapacity(t *testing.T) {
	mem := ai.NewMemory(3)
	for i := 0; i < 5; i++ {
		mem.Add(ai.MemEntry{Type: ai.MemResource, X: i, Y: 0, Resource: "food", Amount: 1.0, Tick: int64(i)})

	entries := mem.Entries()
	if len(entries) != 3 {
		t.Fatalf("expected 3 entries (bounded), got %d", len(entries))

	if entries[0].Tick != 2 {
		t.Errorf("expected oldest remaining tick=2, got %d", entries[0].Tick)
	}
}

Create internal/ai/memory.go:

go
package ai

const staleThreshold = 200 // memories older than this many ticks are ignored

type MemType int

const (
	MemResource MemType = iota
	MemDanger
)

type MemEntry struct {
	Type     MemType
	X        int
	Y        int
	Resource string
	Amount   float64
	Tick     int64
}

type Memory struct {
	entries []MemEntry
	cap     int
}

func NewMemory(capacity int) *Memory {
	return &Memory{
		entries: make([]MemEntry, 0, capacity),
		cap:     capacity,
	}
}

func (m *Memory) Add(entry MemEntry) {
	// Update existing entry for same location+resource
	for i := range m.entries {
		if m.entries[i].X == entry.X && m.entries[i].Y == entry.Y &&
			m.entries[i].Resource == entry.Resource && m.entries[i].Type == entry.Type {
			m.entries[i].Amount = entry.Amount
			m.entries[i].Tick = entry.Tick
			return
		}
	}

	// Add new entry, evict oldest if at capacity
	if len(m.entries) >= m.cap {
		// Remove oldest (index 0)
		m.entries = m.entries[1:]
	}
	m.entries = append(m.entries, entry)
}

func (m *Memory) Entries() []MemEntry {
	return m.entries
}

// FindBestResource returns the best remembered location for a resource.
// Scores by amount * recency. Ignores entries older than staleThreshold ticks.
func (m *Memory) FindBestResource(resource string, agentX, agentY int, currentTick int64) *MemEntry {
	var best *MemEntry
	bestScore := 0.0

	for i := range m.entries {
		e := &m.entries[i]
		if e.Type != MemResource || e.Resource != resource {
			continue
		}
		age := currentTick - e.Tick
		if age > staleThreshold {
			continue
		}

		// Score: amount * recency factor (1.0 for fresh, decays toward 0)
		recency := 1.0 - float64(age)/float64(staleThreshold)
		score := e.Amount * recency
		if score > bestScore {
			bestScore = score
			best = e
		}
	}
	return best
}

go
package ai

import (
	"github.com/alive/worldsim/internal/agent"
	"github.com/alive/worldsim/internal/world"
)

func Perceive(a *agent.Agent, w *world.World, radius int) []*world.Tile {
	var tiles []*world.Tile
	for dy := -radius; dy <= radius; dy++ {
		for dx := -radius; dx <= radius; dx++ {
			tile, err := w.GetTile(a.X+dx, a.Y+dy)
			if err == nil {
				tiles = append(tiles, tile)
			}
		}
	}

	// Record resource locations into memory
	if mem := GetMemory(a); mem != nil {
		for _, tile := range tiles {
			for resID, amount := range tile.Resources {
				if amount > 0 {
					mem.Add(MemEntry{
						Type:     MemResource,
						X:        tile.X,
						Y:        tile.Y,
						Resource: resID,
						Amount:   amount,
						Tick:     w.Tick,
					})
				}
			}
		}
	}

	return tiles
}

func FindNearestResource(a *agent.Agent, tiles []*world.Tile, resourceID string) *world.Tile {
	var best *world.Tile
	bestDist := int(^uint(0) >> 1)

	for _, t := range tiles {
		if t.Resources[resourceID] > 0 {
			dist := abs(t.X-a.X) + abs(t.Y-a.Y)
			if dist < bestDist {
				bestDist = dist
				best = t
			}
		}
	}
	return best
}

func abs(x int) int {
	if x < 0 {
		return -x
	}
	return x
}

// GetMemory retrieves the Memory from an agent, initializing if needed.
func GetMemory(a *agent.Agent) *Memory {
	if a.Memory == nil {
		a.Memory = NewMemory(100)
	}
	mem, ok := a.Memory.(*Memory)
	if !ok {
		mem = NewMemory(100)
		a.Memory = mem
	}
	return mem
}

Read internal/ai/planner.go. Update planEat to consult memory when no food is visible:

go
package ai

import (
	"math/rand"

	"github.com/alive/worldsim/internal/agent"
	"github.com/alive/worldsim/internal/world"
)

const visionRadius = 5

func Plan(a *agent.Agent, w *world.World, tickSeed int64) Action {
	goal := ScoreGoals(a)

	switch goal {
	case GoalEat:
		return planEat(a, w, tickSeed)
	case GoalExplore:
		return planExplore(a, w, tickSeed)
	}
	return Action{Type: ActionIdle}
}

func planEat(a *agent.Agent, w *world.World, tickSeed int64) Action {
	if a.Inventory["food"] > 0 {
		return Action{Type: ActionEat}
	}

	tiles := Perceive(a, w, visionRadius)

	currentTile, _ := w.GetTile(a.X, a.Y)
	if currentTile != nil && currentTile.Resources["food"] > 0 {
		return Action{Type: ActionGather, Resource: "food"}
	}

	foodTile := FindNearestResource(a, tiles, "food")
	if foodTile != nil {
		return Action{Type: ActionMove, TargetX: foodTile.X, TargetY: foodTile.Y}
	}

	// Fallback: check memory for remembered food locations
	mem := GetMemory(a)
	remembered := mem.FindBestResource("food", a.X, a.Y, w.Tick)
	if remembered != nil {
		return Action{Type: ActionMove, TargetX: remembered.X, TargetY: remembered.Y}
	}

	return randomMove(a, w, tickSeed)
}

func planExplore(a *agent.Agent, w *world.World, tickSeed int64) Action {
	return randomMove(a, w, tickSeed)
}

func randomMove(a *agent.Agent, w *world.World, tickSeed int64) Action {
	rng := rand.New(rand.NewSource(tickSeed + int64(a.X*1000+a.Y)))
	dirs := [][2]int{{0, -1}, {0, 1}, {-1, 0}, {1, 0}}
	dir := dirs[rng.Intn(len(dirs))]
	return Action{
		Type:    ActionMove,
		TargetX: a.X + dir[0],
		TargetY: a.Y + dir[1],
	}
}
