For agentic workers: REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.

package event

import "fmt"

type EventType int

const (
	EvtGather EventType = iota
	EvtEat
	EvtCraft
	EvtTrade
	EvtDeath
	EvtMove
)

var eventNames = map[EventType]string{
	EvtGather: "GATHER",
	EvtEat:    "EAT",
	EvtCraft:  "CRAFT",
	EvtTrade:  "TRADE",
	EvtDeath:  "DEATH",
	EvtMove:   "MOVE",
}

type Event struct {
	Type    EventType `json:"type"`
	Tick    int64     `json:"tick"`
	AgentID string    `json:"agent_id"`
	Message string    `json:"message"`
}

func (e Event) String() string {
	name := eventNames[e.Type]
	return fmt.Sprintf("[%d] %s %s: %s", e.Tick, name, e.AgentID, e.Message)
}

type Bus struct {
	events []Event
	cap    int
}

func NewBus(capacity int) *Bus {
	return &Bus{
		events: make([]Event, 0, capacity),
		cap:    capacity,
	}
}

func (b *Bus) Emit(e Event) {
	if len(b.events) >= b.cap {
		// Drop oldest
		b.events = b.events[1:]
	}
	b.events = append(b.events, e)
}

func (b *Bus) Drain() []Event {
	result := make([]Event, len(b.events))
	copy(result, b.events)
	b.events = b.events[:0]
	return result
}

go
package main

import (
	"context"
	"encoding/json"
	"flag"
	"fmt"
	"log"
	"os"
	"strings"

	tea "github.com/charmbracelet/bubbletea"
	"github.com/alive/worldsim/internal/cache"
)

const (
	inspectorWidth = 32
	eventLogHeight = 8
	maxEvents      = 50
)

type model struct {
	state       *cache.WorldState
	msgCh       <-chan string
	width       int
	height      int
	err         error
	selectedIdx int
	eventLog    []cache.EventMsg
}

type stateMsg cache.WorldState
type errMsg error

func waitForState(ch <-chan string) tea.Cmd {
	return func() tea.Msg {
		msg, ok := <-ch
		if !ok {
			return errMsg(fmt.Errorf("channel closed"))
		}
		var state cache.WorldState
		if err := json.Unmarshal([]byte(msg), &state); err != nil {
			return errMsg(err)
		}
		return stateMsg(state)
	}
}

func (m model) Init() tea.Cmd {
	return waitForState(m.msgCh)
}

func (m model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.KeyMsg:
		switch msg.String() {
		case "q", "ctrl+c":
			return m, tea.Quit
		case "j", "down":
			if m.state != nil && m.selectedIdx < len(m.state.Agents)-1 {
				m.selectedIdx++
			}
		case "k", "up":
			if m.selectedIdx > 0 {
				m.selectedIdx--
			}
		}
	case tea.WindowSizeMsg:
		m.width = msg.Width
		m.height = msg.Height
	case stateMsg:
		state := cache.WorldState(msg)
		m.state = &state
		// Append new events to log
		for _, e := range state.Events {
			m.eventLog = append(m.eventLog, e)
		}
		if len(m.eventLog) > maxEvents {
			m.eventLog = m.eventLog[len(m.eventLog)-maxEvents:]
		}
		// Keep selected index in bounds
		if m.selectedIdx >= len(state.Agents) {
			m.selectedIdx = len(state.Agents) - 1
		}
		if m.selectedIdx < 0 {
			m.selectedIdx = 0
		}
		return m, waitForState(m.msgCh)
	case errMsg:
		m.err = msg
		return m, nil
	}
	return m, nil
}

func (m model) View() string {
	if m.err != nil {
		return fmt.Sprintf("Error: %v\nPress q to quit.", m.err)
	}
	if m.state == nil {
		return "Waiting for world state...\nPress q to quit."
	}

	var b strings.Builder

	gridWidth := m.width - inspectorWidth - 3
	if gridWidth < 20 {
		gridWidth = 20
	}
	gridHeight := m.height - eventLogHeight - 4
	if gridHeight < 5 {
		gridHeight = 5
	}

	// Header
	header := fmt.Sprintf(" WorldSim | Tick: %d | Alive: %d Dead: %d ",
		m.state.Tick, m.state.AliveCount, m.state.DeadCount)
	b.WriteString(header)
	if len(header) < m.width {
		b.WriteString(strings.Repeat(" ", m.width-len(header)))
	}
	b.WriteRune('\n')
	b.WriteString(strings.Repeat("-", m.width))
	b.WriteRune('\n')

	// Build agent position map
	agentMap := make(map[[2]int]int)
	for _, a := range m.state.Agents {
		agentMap[[2]int{a.X, a.Y}]++
	}

	// Scale factors
	scaleX := 1
	scaleY := 1
	if m.state.Width > gridWidth {
		scaleX = (m.state.Width + gridWidth - 1) / gridWidth
	}
	if m.state.Height > gridHeight {
		scaleY = (m.state.Height + gridHeight - 1) / gridHeight
	}
	displayW := m.state.Width / scaleX
	displayH := m.state.Height / scaleY
	if displayW > gridWidth {
		displayW = gridWidth
	}
	if displayH > gridHeight {
		displayH = gridHeight
	}

	// Build inspector lines
	inspLines := m.renderInspector()

	// Render grid + inspector side by side
	for dy := 0; dy < displayH || dy < len(inspLines); dy++ {
		// Grid column
		if dy < displayH {
			for dx := 0; dx < displayW; dx++ {
				hasAgent := false
				for sy := 0; sy < scaleY; sy++ {
					for sx := 0; sx < scaleX; sx++ {
						wx := dx*scaleX + sx
						wy := dy*scaleY + sy
						if agentMap[[2]int{wx, wy}] > 0 {
							hasAgent = true
						}
					}
				}
				if hasAgent {
					b.WriteRune('@')
				} else {
					b.WriteRune('.')
				}
			}
			// Pad to gridWidth
			if displayW < gridWidth {
				b.WriteString(strings.Repeat(" ", gridWidth-displayW))
			}
		} else {
			b.WriteString(strings.Repeat(" ", gridWidth))
		}

		b.WriteString(" | ")

		// Inspector column
		if dy < len(inspLines) {
			line := inspLines[dy]
			b.WriteString(line)
			if len(line) < inspectorWidth {
				b.WriteString(strings.Repeat(" ", inspectorWidth-len(line)))
			}
		} else {
			b.WriteString(strings.Repeat(" ", inspectorWidth))
		}
		b.WriteRune('\n')
	}

	// Event log separator
	b.WriteString(strings.Repeat("-", m.width))
	b.WriteRune('\n')
	b.WriteString(" Event Log\n")

	// Show last N events (newest first)
	eventLines := eventLogHeight - 2
	if eventLines < 1 {
		eventLines = 1
	}
	start := len(m.eventLog) - eventLines
	if start < 0 {
		start = 0
	}
	for i := len(m.eventLog) - 1; i >= start; i-- {
		e := m.eventLog[i]
		line := fmt.Sprintf(" [%d] %s %s: %s", e.Tick, e.Type, e.AgentID, e.Message)
		if len(line) > m.width-1 {
			line = line[:m.width-1]
		}
		b.WriteString(line)
		b.WriteRune('\n')
	}

	// Status bar
	b.WriteString(strings.Repeat("-", m.width))
	b.WriteRune('\n')
	b.WriteString(" [j/k] select agent  [q] quit")

	return b.String()
}

func (m model) renderInspector() []string {
	if m.state == nil || len(m.state.Agents) == 0 {
		return []string{"No agents"}
	}

	a := m.state.Agents[m.selectedIdx]
	lines := []string{
		fmt.Sprintf("Agent Inspector [%d/%d]", m.selectedIdx+1, len(m.state.Agents)),
		"",
		fmt.Sprintf("%s (%d,%d)", a.ID, a.X, a.Y),
		fmt.Sprintf("HP:%.0f  EN:%.0f  HU:%.0f", a.Health, a.Energy, a.Hunger),
		"",
		"Inventory:",
	}

	if len(a.Inventory) == 0 {
		lines = append(lines, "  (empty)")
	} else {
		for res, qty := range a.Inventory {
			if qty > 0.01 {
				lines = append(lines, fmt.Sprintf("  %-10s %.1f", res, qty))
			}
		}
	}

	lines = append(lines, "", "Skills:")
	if len(a.Skills) == 0 {
		lines = append(lines, "  (none)")
	} else {
		for skill, level := range a.Skills {
			if level > 0.001 {
				lines = append(lines, fmt.Sprintf("  %-10s %.2f", skill, level))
			}
		}
	}

	return lines
}

func main() {
	redisAddr := flag.String("redis", "localhost:6379", "redis address")
	flag.Parse()

	ctx := context.Background()
	rc, err := cache.NewRedisCache(ctx, *redisAddr)
	if err != nil {
		log.Fatalf("connect to redis: %v", err)
	}
	defer rc.Close()

	initialState, _ := rc.GetState(ctx)

	ch, err := rc.Subscribe(ctx)
	if err != nil {
		log.Fatalf("subscribe: %v", err)
	}

	m := model{
		state:    initialState,
		msgCh:    ch,
		width:    120,
		height:   40,
		eventLog: make([]cache.EventMsg, 0),
	}

	p := tea.NewProgram(m, tea.WithAltScreen())
	if _, err := p.Run(); err != nil {
		fmt.Fprintf(os.Stderr, "Error: %v\n", err)
		os.Exit(1)
	}
}
