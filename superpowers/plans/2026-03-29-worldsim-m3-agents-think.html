type Agent struct {
	ID        string             `json:"id"`
	X         int                `json:"x"`
	Y         int                `json:"y"`
	Health    float64            `json:"health"`
	Energy    float64            `json:"energy"`
	Hunger    float64            `json:"hunger"`
	Alive     bool               `json:"alive"`
	Inventory map[string]float64 `json:"inventory"`
}

func NewAgent(id string, x, y int) *Agent {
	return &Agent{
		ID:        id,
		X:         x,
		Y:         y,
		Health:    100,
		Energy:    100,
		Hunger:    0,
		Alive:     true,
		Inventory: make(map[string]float64),
	}
}

func SpawnAgents(count, worldWidth, worldHeight int, seed int64) []*Agent {
	rng := rand.New(rand.NewSource(seed))
	agents := make([]*Agent, count)
	for i := 0; i < count; i++ {
		x := rng.Intn(worldWidth)
		y := rng.Intn(worldHeight)
		agents[i] = NewAgent(fmt.Sprintf("agent-%03d", i), x, y)
	}
	return agents
}

package ai

import (
	"github.com/alive/worldsim/internal/agent"
	"github.com/alive/worldsim/internal/world"
)

// Perceive returns all tiles within the given radius of the agent.
func Perceive(a *agent.Agent, w *world.World, radius int) []*world.Tile {
	var tiles []*world.Tile
	for dy := -radius; dy <= radius; dy++ {
		for dx := -radius; dx <= radius; dx++ {
			tile, err := w.GetTile(a.X+dx, a.Y+dy)
			if err == nil {
				tiles = append(tiles, tile)
			}
		}
	}
	return tiles
}

// FindNearestResource returns the closest tile with the given resource, or nil.
func FindNearestResource(a *agent.Agent, tiles []*world.Tile, resourceID string) *world.Tile {
	var best *world.Tile
	bestDist := int(^uint(0) >> 1) // max int

	for _, t := range tiles {
		if t.Resources[resourceID] > 0 {
			dist := abs(t.X-a.X) + abs(t.Y-a.Y) // Manhattan distance
			if dist < bestDist {
				bestDist = dist
				best = t
			}
		}
	}
	return best
}

func abs(x int) int {
	if x < 0 {
		return -x
	}
	return x
}

package ai

import (
	"github.com/alive/worldsim/internal/agent"
	"github.com/alive/worldsim/internal/world"
)

// ActionType identifies what kind of action this is.
type ActionType int

const (
	ActionMove ActionType = iota
	ActionGather
	ActionEat
	ActionIdle
)

// Action represents a planned action for an agent.
type Action struct {
	Type   ActionType
	TargetX int
	TargetY int
	Resource string // for Gather: which resource
}

// Execute applies the action to the agent and world. Returns true if successful.
func (a *Action) Execute(ag *agent.Agent, w *world.World) bool {
	switch a.Type {
	case ActionMove:
		return executeMove(ag, w, a.TargetX, a.TargetY)
	case ActionGather:
		return executeGather(ag, w, a.Resource)
	case ActionEat:
		return executeEat(ag)
	case ActionIdle:
		return true
	}
	return false
}

func executeMove(ag *agent.Agent, w *world.World, tx, ty int) bool {
	// Move one step toward target
	dx := sign(tx - ag.X)
	dy := sign(ty - ag.Y)
	newX := ag.X + dx
	newY := ag.Y + dy
	if newX >= 0 && newX < w.Width && newY >= 0 && newY < w.Height {
		ag.X = newX
		ag.Y = newY
		ag.Energy -= 1
		return true
	}
	return false
}

func executeGather(ag *agent.Agent, w *world.World, resource string) bool {
	tile, err := w.GetTile(ag.X, ag.Y)
	if err != nil {
		return false
	}
	available := tile.Resources[resource]
	if available <= 0 {
		return false
	}
	// Gather up to 1.0 units per tick
	amount := 1.0
	if available < amount {
		amount = available
	}
	tile.Resources[resource] -= amount
	ag.Inventory[resource] += amount
	ag.Energy -= 0.5
	return true
}

func executeEat(ag *agent.Agent) bool {
	food := ag.Inventory["food"]
	if food <= 0 {
		return false
	}
	// Consume up to 1.0 food, reduce hunger by 30
	amount := 1.0
	if food < amount {
		amount = food
	}
	ag.Inventory["food"] -= amount
	ag.Hunger -= 30 * amount
	if ag.Hunger < 0 {
		ag.Hunger = 0
	}
	return true
}

func sign(x int) int {
	if x > 0 {
		return 1
	}
	if x < 0 {
		return -1
	}
	return 0
}

import "github.com/alive/worldsim/internal/agent"

// GoalType identifies what the agent wants.
type GoalType int

const (
	GoalEat GoalType = iota
	GoalExplore
)

// ScoreGoals returns the highest-priority goal for the agent.
func ScoreGoals(a *agent.Agent) GoalType {
	// Hungry agents prioritize eating
	if a.Hunger > 30 {
		return GoalEat
	}
	return GoalExplore
}

package ai

import (
	"math/rand"

	"github.com/alive/worldsim/internal/agent"
	"github.com/alive/worldsim/internal/world"
)

const visionRadius = 5

// Plan decides the next action for an agent based on its current state and surroundings.
func Plan(a *agent.Agent, w *world.World, tickSeed int64) Action {
	goal := ScoreGoals(a)

	switch goal {
	case GoalEat:
		return planEat(a, w, tickSeed)
	case GoalExplore:
		return planExplore(a, w, tickSeed)
	}
	return Action{Type: ActionIdle}
}

func planEat(a *agent.Agent, w *world.World, tickSeed int64) Action {
	// If we have food, eat it
	if a.Inventory["food"] > 0 {
		return Action{Type: ActionEat}
	}

	// Look for food to gather
	tiles := Perceive(a, w, visionRadius)

	// Check current tile first
	currentTile, _ := w.GetTile(a.X, a.Y)
	if currentTile != nil && currentTile.Resources["food"] > 0 {
		return Action{Type: ActionGather, Resource: "food"}
	}

	// Find nearest food tile
	foodTile := FindNearestResource(a, tiles, "food")
	if foodTile != nil {
		return Action{Type: ActionMove, TargetX: foodTile.X, TargetY: foodTile.Y}
	}

	// No food visible, explore randomly to find some
	return randomMove(a, w, tickSeed)
}

func planExplore(a *agent.Agent, w *world.World, tickSeed int64) Action {
	return randomMove(a, w, tickSeed)
}

func randomMove(a *agent.Agent, w *world.World, tickSeed int64) Action {
	rng := rand.New(rand.NewSource(tickSeed + int64(a.X*1000+a.Y)))
	dirs := [][2]int{{0, -1}, {0, 1}, {-1, 0}, {1, 0}}
	dir := dirs[rng.Intn(len(dirs))]
	return Action{
		Type:    ActionMove,
		TargetX: a.X + dir[0],
		TargetY: a.Y + dir[1],
	}
}

package simulation

import (
	"sync"

	"github.com/alive/worldsim/internal/agent"
	"github.com/alive/worldsim/internal/ai"
	"github.com/alive/worldsim/internal/world"
)

const (
	hungerPerTick  = 0.5
	energyPerTick  = 0.2
	starvationHP   = 2.0 // HP lost per tick when hunger >= 100
)

type Simulation struct {
	world      *world.World
	agents     []*agent.Agent
	numWorkers int
}

func New(w *world.World, agents []*agent.Agent, numWorkers int) *Simulation {
	return &Simulation{
		world:      w,
		agents:     agents,
		numWorkers: numWorkers,
	}
}

func (s *Simulation) World() *world.World {
	return s.world
}

func (s *Simulation) Agents() []*agent.Agent {
	return s.agents
}

func (s *Simulation) AliveAgents() []*agent.Agent {
	var alive []*agent.Agent
	for _, a := range s.agents {
		if a.Alive {
			alive = append(alive, a)
		}
	}
	return alive
}

func (s *Simulation) Tick(tickNum int64) {
	s.world.Tick = tickNum

	// Phase 1: AI planning in parallel (read-only on world)
	actions := make([]ai.Action, len(s.agents))

	var wg sync.WaitGroup
	work := make(chan int, len(s.agents))

	for w := 0; w < s.numWorkers; w++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			for idx := range work {
				a := s.agents[idx]
				if !a.Alive {
					actions[idx] = ai.Action{Type: ai.ActionIdle}
					continue
				}
				actions[idx] = ai.Plan(a, s.world, tickNum)
			}
		}()
	}

	for i := range s.agents {
		work <- i
	}
	close(work)
	wg.Wait()

	// Phase 2: Execute actions sequentially
	for i, action := range actions {
		a := s.agents[i]
		if !a.Alive {
			continue
		}
		action.Execute(a, s.world)
	}

	// Phase 3: Apply hunger, energy decay, starvation
	for _, a := range s.agents {
		if !a.Alive {
			continue
		}
		a.Hunger += hungerPerTick
		a.Energy -= energyPerTick

		// Clamp energy
		if a.Energy < 0 {
			a.Energy = 0
		}

		// Starvation damage
		if a.Hunger >= 100 {
			a.Health -= starvationHP
			if a.Health <= 0 {
				a.Health = 0
				a.Alive = false
			}
		}
	}
}

func makeTestWorld() *world.World {
	wc := config.WorldConfig{Width: 16, Height: 16, Seed: 42}
	terrains := []config.TerrainDef{
		{ID: "grass", MovementCost: 1.0, Resources: []string{"food"}},
	}
	resources := []config.ResourceDef{
		{ID: "food", Category: "raw", Weight: 1, TerrainSources: []string{"grass"}, RegenRate: 0.3},
	}
	return world.GenerateWorld(wc, terrains, resources)
}

func TestNewSimulation(t *testing.T) {
	w := makeTestWorld()
	agents := agent.SpawnAgents(5, w.Width, w.Height, 42)
	sim := simulation.New(w, agents, 4)

	if sim.World() != w {
		t.Error("expected same world reference")
	}
	if len(sim.Agents()) != 5 {
		t.Errorf("expected 5 agents, got %d", len(sim.Agents()))
	}
}

func TestSimulation_Tick_AgentsMove(t *testing.T) {
	w := makeTestWorld()
	agents := agent.SpawnAgents(10, w.Width, w.Height, 42)

	type pos struct{ x, y int }
	initial := make([]pos, len(agents))
	for i, a := range agents {
		initial[i] = pos{a.X, a.Y}
	}

	sim := simulation.New(w, agents, 4)
	sim.Tick(1)

	moved := 0
	for i, a := range agents {
		if a.X != initial[i].x || a.Y != initial[i].y {
			moved++
		}
	}
	if moved == 0 {
		t.Error("expected at least some agents to move after a tick")
	}
}

func TestSimulation_Tick_AgentsStayInBounds(t *testing.T) {
	w := makeTestWorld()
	agents := agent.SpawnAgents(20, w.Width, w.Height, 42)
	sim := simulation.New(w, agents, 4)

	for tick := int64(1); tick <= 50; tick++ {
		sim.Tick(tick)
		for _, a := range agents {
			if a.Alive && (a.X < 0 || a.X >= w.Width || a.Y < 0 || a.Y >= w.Height) {
				t.Fatalf("agent %s out of bounds at (%d,%d) on tick %d",
					a.ID, a.X, a.Y, tick)
			}
		}
	}
}

func TestSimulation_Tick_WorldTickIncremented(t *testing.T) {
	w := makeTestWorld()
	agents := agent.SpawnAgents(3, w.Width, w.Height, 42)
	sim := simulation.New(w, agents, 4)

	sim.Tick(1)
	if w.Tick != 1 {

	}
}

func TestSimulation_HungerIncreases(t *testing.T) {
	w := makeTestWorld()
	agents := []*agent.Agent{agent.NewAgent("test", 8, 8)}
	sim := simulation.New(w, agents, 1)

	sim.Tick(1)

	if agents[0].Hunger <= 0 {
		t.Error("expected hunger to increase after tick")
	}
}

func TestSimulation_StarvationKills(t *testing.T) {
	w := makeTestWorld()
	a := agent.NewAgent("test", 8, 8)
	a.Hunger = 99 // will cross 100 next tick
	a.Health = 3  // low health, will die quickly
	agents := []*agent.Agent{a}

	sim := simulation.New(w, agents, 1)

	// Run enough ticks to kill the agent
	for tick := int64(1); tick <= 10; tick++ {
		sim.Tick(tick)
		if !a.Alive {
			break
		}
	}

	if a.Alive {
		t.Errorf("expected agent to die from starvation, health=%f hunger=%f", a.Health, a.Hunger)
	}
}

func TestSimulation_AliveAgents(t *testing.T) {
	w := makeTestWorld()
	agents := []*agent.Agent{
		agent.NewAgent("a1", 1, 1),
		agent.NewAgent("a2", 2, 2),
	}
	agents[1].Alive = false

	sim := simulation.New(w, agents, 1)
	alive := sim.AliveAgents()

	if len(alive) != 1 {
		t.Errorf("expected 1 alive agent, got %d", len(alive))
	}
	if alive[0].ID != "a1" {
		t.Errorf("expected alive agent to be a1, got %s", alive[0].ID)
	}
}

func TestSimulation_HungryAgent_GathersFood(t *testing.T) {
	w := makeTestWorld()
	a := agent.NewAgent("test", 8, 8)
	a.Hunger = 60

	// Ensure there's food on the agent's tile
	tile, _ := w.GetTile(8, 8)
	tile.Resources["food"] = 10.0

	agents := []*agent.Agent{a}
	sim := simulation.New(w, agents, 1)

	// Run a few ticks - agent should gather and eat
	for tick := int64(1); tick <= 5; tick++ {
		sim.Tick(tick)
	}

	// Agent should have gathered some food
	if a.Inventory["food"] == 0 && tile.Resources["food"] == 10.0 {
		t.Error("expected agent to gather some food")
	}
}
