WorldSim M1: World Exists -- Implementation Plan ​

For agentic workers: REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.

Goal: Generate a persistent tile-based world from YAML config, save it to Postgres, and reload it on restart.

Architecture: Single Go binary reading YAML configs (terrains, resources, world settings), generating a 2D tile grid using Perlin noise, persisting snapshots to Postgres via golang-migrate managed schemas. Docker Compose provides Postgres and Redis (Redis wired up but not used until M2).

Tech Stack: Go 1.22+, Postgres 16, Redis 7, Docker Compose, golang-migrate, gopkg.in/yaml.v3, github.com/aquilax/go-perlin

package config

import (
	"fmt"
	"os"

	"gopkg.in/yaml.v3"
)

type terrainsFile struct {
	Terrains []TerrainDef `yaml:"terrains"`
}

type resourcesFile struct {
	Resources []ResourceDef `yaml:"resources"`
}

func LoadTerrains(path string) ([]TerrainDef, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("read terrains file: %w", err)
	}
	var f terrainsFile
	if err := yaml.Unmarshal(data, &f); err != nil {
		return nil, fmt.Errorf("parse terrains file: %w", err)
	}
	return f.Terrains, nil
}

func LoadResources(path string) ([]ResourceDef, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("read resources file: %w", err)
	}
	var f resourcesFile
	if err := yaml.Unmarshal(data, &f); err != nil {
		return nil, fmt.Errorf("parse resources file: %w", err)
	}
	return f.Resources, nil
}

func LoadWorld(path string) (WorldConfig, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return WorldConfig{}, fmt.Errorf("read world config: %w", err)
	}
	var wc WorldConfig
	if err := yaml.Unmarshal(data, &wc); err != nil {
		return WorldConfig{}, fmt.Errorf("parse world config: %w", err)
	}
	return wc, nil
}

Create internal/world/world.go:

package world

import "fmt"

type World struct {
	Width  int      `json:"width"`
	Height int      `json:"height"`
	Tick   int64    `json:"tick"`
	Seed   int64    `json:"seed"`
	Tiles  [][]Tile `json:"tiles"`
}

func NewWorld(width, height int, seed int64) *World {
	tiles := make([][]Tile, height)
	for y := range tiles {
		tiles[y] = make([]Tile, width)
		for x := range tiles[y] {
			tiles[y][x] = Tile{
				X:         x,
				Y:         y,
				Resources: make(map[string]float64),
			}
		}
	}
	return &World{
		Width:  width,
		Height: height,
		Seed:   seed,
		Tiles:  tiles,
	}
}

func (w *World) GetTile(x, y int) (*Tile, error) {
	if x < 0 || x >= w.Width || y < 0 || y >= w.Height {
		return nil, fmt.Errorf("tile (%d, %d) out of bounds [%d x %d]", x, y, w.Width, w.Height)
	}
	return &w.Tiles[y][x], nil
}

Create internal/world/generate.go:

go
package world

import (
	"github.com/alive/worldsim/internal/config"
	"github.com/aquilax/go-perlin"
)

const (
	perlinAlpha = 2.0
	perlinBeta  = 2.0
	perlinN     = 3
	noiseScale  = 0.05
)

func GenerateWorld(wc config.WorldConfig, terrains []config.TerrainDef, resources []config.ResourceDef) *World {
	w := NewWorld(wc.Width, wc.Height, wc.Seed)
	p := perlin.NewPerlin(perlinAlpha, perlinBeta, perlinN, wc.Seed)

	resourcesByTerrain := buildResourceMap(terrains, resources)

	for y := 0; y < wc.Height; y++ {
		for x := 0; x < wc.Width; x++ {
			noise := p.Noise2D(float64(x)*noiseScale, float64(y)*noiseScale)
			terrain := pickTerrain(noise, terrains)

			tile := &w.Tiles[y][x]
			tile.Terrain = terrain.ID

			for _, res := range resourcesByTerrain[terrain.ID] {
				tile.Resources[res.ID] = res.RegenRate * 10
			}
		}
	}

	return w
}

func pickTerrain(noise float64, terrains []config.TerrainDef) config.TerrainDef {
	// Perlin noise returns roughly -1 to 1.
	// Divide the range evenly among terrain types, ordered as given.
	n := len(terrains)
	// Normalize to 0..1
	norm := (noise + 1.0) / 2.0
	if norm >= 1.0 {
		norm = 0.999
	}
	idx := int(norm * float64(n))
	if idx < 0 {
		idx = 0
	}
	if idx >= n {
		idx = n - 1
	}
	return terrains[idx]
}

func buildResourceMap(terrains []config.TerrainDef, resources []config.ResourceDef) map[string][]config.ResourceDef {
	result := make(map[string][]config.ResourceDef)
	for _, td := range terrains {
		terrainRes := make([]config.ResourceDef, 0)
		for _, rd := range resources {
			for _, src := range rd.TerrainSources {
				if src == td.ID {
					terrainRes = append(terrainRes, rd)
					break
				}
			}
		}
		result[td.ID] = terrainRes
	}
	return result
}

sql
CREATE TABLE IF NOT EXISTS world_snapshots (
    id          BIGSERIAL PRIMARY KEY,
    tick        BIGINT NOT NULL,
    seed        BIGINT NOT NULL,
    width       INT NOT NULL,
    height      INT NOT NULL,
    world_data  JSONB NOT NULL,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()

import (

	data, err := json.Marshal(w)
	if err != nil {
		return nil, fmt.Errorf("serialize world: %w", err)
	}
	return data, nil
}

func DeserializeWorld(data []byte) (*world.World, error) {
	var w world.World
	if err := json.Unmarshal(data, &w); err != nil {
		return nil, fmt.Errorf("deserialize world: %w", err)
	}
	return &w, nil
}

go
package storage

import (
	"context"
	"database/sql"
	"fmt"

	"github.com/golang-migrate/migrate/v4"
	migratepostgres "github.com/golang-migrate/migrate/v4/database/postgres"
	_ "github.com/golang-migrate/migrate/v4/source/file"
	_ "github.com/lib/pq"

	"github.com/alive/worldsim/internal/world"
)

type PostgresStore struct {
	db *sql.DB
}

func NewPostgresStore(ctx context.Context, dsn string, migrationsPath string) (*PostgresStore, error) {
	db, err := sql.Open("postgres", dsn)
	if err != nil {
		return nil, fmt.Errorf("open postgres: %w", err)
	}
	if err := db.PingContext(ctx); err != nil {
		return nil, fmt.Errorf("ping postgres: %w", err)
	}

	if err := runMigrations(db, migrationsPath); err != nil {
		return nil, fmt.Errorf("run migrations: %w", err)
	}

	return &PostgresStore{db: db}, nil
}

func (s *PostgresStore) Close() error {
	return s.db.Close()
}

func (s *PostgresStore) DB() *sql.DB {
	return s.db
}

func (s *PostgresStore) SaveSnapshot(ctx context.Context, w *world.World) error {
	data, err := SerializeWorld(w)
	if err != nil {
		return err
	}

	_, err = s.db.ExecContext(ctx,
		`INSERT INTO world_snapshots (tick, seed, width, height, world_data) VALUES ($1, $2, $3, $4, $5)`,
		w.Tick, w.Seed, w.Width, w.Height, data,
	)
	if err != nil {
		return fmt.Errorf("insert snapshot: %w", err)
	}
	return nil
}

func (s *PostgresStore) LoadLatestSnapshot(ctx context.Context) (*world.World, error) {
	var data []byte
	err := s.db.QueryRowContext(ctx,
		`SELECT world_data FROM world_snapshots ORDER BY tick DESC, id DESC LIMIT 1`,
	).Scan(&data)
	if err == sql.ErrNoRows {
		return nil, nil
	}
	if err != nil {
		return nil, fmt.Errorf("query latest snapshot: %w", err)
	}

	w, err := DeserializeWorld(data)
	if err != nil {
		return nil, err
	}
	return w, nil
}

func runMigrations(db *sql.DB, migrationsPath string) error {
	driver, err := migratepostgres.WithInstance(db, &migratepostgres.Config{})
	if err != nil {
		return fmt.Errorf("create migrate driver: %w", err)
	}

	m, err := migrate.NewWithDatabaseInstance(
		"file://"+migrationsPath,
		"postgres",
		driver,
	)
	if err != nil {
		return fmt.Errorf("create migrate instance: %w", err)
	}

	if err := m.Up(); err != nil && err != migrate.ErrNoChange {
		return fmt.Errorf("migrate up: %w", err)
	}
	return nil
}

package main

import (
	"context"
	"flag"
	"fmt"
	"log"
	"os"

	"github.com/alive/worldsim/internal/config"
	"github.com/alive/worldsim/internal/storage"
	"github.com/alive/worldsim/internal/world"
)

func main() {
	configDir := flag.String("config", "config", "path to config directory")
	migrationsDir := flag.String("migrations", "migrations", "path to migrations directory")
	dsn := flag.String("dsn", "postgres://worldsim:worldsim@localhost:5432/worldsim?sslmode=disable", "postgres connection string")
	flag.Parse()

	ctx := context.Background()

	// Load configs
	terrains, err := config.LoadTerrains(*configDir + "/terrains.yaml")
	if err != nil {
		log.Fatalf("load terrains: %v", err)
	}
	fmt.Fprintf(os.Stderr, "Loaded %d terrain types\n", len(terrains))

	resources, err := config.LoadResources(*configDir + "/resources.yaml")
	if err != nil {
		log.Fatalf("load resources: %v", err)
	}
	fmt.Fprintf(os.Stderr, "Loaded %d resource types\n", len(resources))

	wc, err := config.LoadWorld(*configDir + "/world.yaml")
	if err != nil {
		log.Fatalf("load world config: %v", err)
	}
	fmt.Fprintf(os.Stderr, "World config: %dx%d, seed=%d\n", wc.Width, wc.Height, wc.Seed)

	// Connect to Postgres
	store, err := storage.NewPostgresStore(ctx, *dsn, *migrationsDir)
	if err != nil {
		log.Fatalf("connect to postgres: %v", err)
	}
	defer store.Close()
	fmt.Fprintln(os.Stderr, "Connected to Postgres")

	// Try to load existing world
	w, err := store.LoadLatestSnapshot(ctx)
	if err != nil {
		log.Fatalf("load snapshot: %v", err)
	}

	if w != nil {
		fmt.Fprintf(os.Stderr, "Loaded world from snapshot at tick %d\n", w.Tick)
	} else {
		fmt.Fprintln(os.Stderr, "No snapshot found, generating new world...")
		w = world.GenerateWorld(wc, terrains, resources)
		fmt.Fprintf(os.Stderr, "Generated %dx%d world with seed %d\n", w.Width, w.Height, w.Seed)

		if err := store.SaveSnapshot(ctx, w); err != nil {
			log.Fatalf("save snapshot: %v", err)
		}
		fmt.Fprintln(os.Stderr, "Saved initial snapshot to Postgres")
	}

	// Print world stats
	printWorldStats(w, terrains)
}

func printWorldStats(w *world.World, terrains []config.TerrainDef) {
	terrainCounts := make(map[string]int)
	totalResources := make(map[string]float64)

	for y := 0; y < w.Height; y++ {
		for x := 0; x < w.Width; x++ {
			tile, _ := w.GetTile(x, y)
			terrainCounts[tile.Terrain]++
			for res, qty := range tile.Resources {
				totalResources[res] += qty
			}
		}
	}

	fmt.Printf("\n=== World Stats (tick %d) ===\n", w.Tick)
	fmt.Printf("Size: %d x %d (%d tiles)\n\n", w.Width, w.Height, w.Width*w.Height)

	fmt.Println("Terrain distribution:")
	for _, td := range terrains {
		count := terrainCounts[td.ID]
		pct := float64(count) / float64(w.Width*w.Height) * 100
		fmt.Printf("  %-10s %5d tiles (%.1f%%)\n", td.ID, count, pct)
	}

	fmt.Println("\nTotal resources:")
	for res, qty := range totalResources {
		fmt.Printf("  %-10s %.1f\n", res, qty)
	}
	fmt.Println()
}
