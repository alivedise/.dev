WorldSim M6: Agents Trade -- Implementation Plan ​

For agentic workers: REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.

Goal: Agents barter resources with adjacent agents based on subjective value (personal scarcity). Trust scores track trade history. Agents specialize: one gathers wood, another gathers food, they trade complementary resources.

Architecture: New internal/economy/trade.go for trade logic. Add Trust map to Agent. Add GoalTrade and ActionTrade. Trade resolution happens in the simulation's sequential phase after individual actions. Each tick, adjacent agents with complementary needs can exchange resources.

Tech Stack: Pure Go, no new dependencies.

Create internal/economy/trade.go:

go
package economy

import (
	"math"

	"github.com/alive/worldsim/internal/agent"
)

const (
	tradeRadius   = 1   // must be adjacent
	tradeAmount   = 2.0 // units exchanged per trade
	trustGain     = 5.0 // trust gained per successful trade
	minTradeGain  = 0.5 // minimum subjective value difference to bother trading
)

// SubjectiveValue returns how much an agent values a resource.
// Scarce resources are valued higher. Uses inverse of current stock.
func SubjectiveValue(a *agent.Agent, resource string) float64 {
	stock := a.Inventory[resource]
	// Value = 10 / (stock + 1) -- high when scarce, low when abundant
	return 10.0 / (stock + 1.0)
}

// FindTradePartner finds the best adjacent agent to trade with.
// Returns the partner, what to offer (our surplus), and what we want (our deficit).
func FindTradePartner(a *agent.Agent, others []*agent.Agent) (*agent.Agent, string, string) {
	// Find our most abundant and most scarce resources
	offer, want := bestTradeResources(a)
	if offer == "" || want == "" {
		return nil, "", ""
	}

	var bestPartner *agent.Agent
	bestScore := 0.0

	for _, other := range others {
		if !other.Alive || other.ID == a.ID {
			continue
		}
		// Must be adjacent
		if abs(a.X-other.X) > tradeRadius || abs(a.Y-other.Y) > tradeRadius {
			continue
		}

		// Check if trade is mutually beneficial:
		// We offer what's cheap for us, want what's expensive for us.
		// Partner should value our offer and have surplus of what we want.
		ourOfferVal := SubjectiveValue(a, offer)
		ourWantVal := SubjectiveValue(a, want)
		theirOfferVal := SubjectiveValue(other, want) // they'd give us what we want
		theirWantVal := SubjectiveValue(other, offer)  // they want what we offer

		// Both sides must gain: we value what we get more than what we give,
		// and they value what they get more than what they give.
		ourGain := ourWantVal - ourOfferVal
		theirGain := theirWantVal - theirOfferVal

		if ourGain < minTradeGain || theirGain < minTradeGain {
			continue
		}

		// Must have enough to trade
		if a.Inventory[offer] < tradeAmount || other.Inventory[want] < tradeAmount {
			continue
		}

		score := ourGain + theirGain
		if score > bestScore {
			bestScore = score
			bestPartner = other
		}
	}

	return bestPartner, offer, want
}

// ExecuteTrade performs a barter exchange between two agents.
func ExecuteTrade(giver, receiver *agent.Agent, offerRes, wantRes string, amount float64) {
	if giver.Inventory[offerRes] < amount || receiver.Inventory[wantRes] < amount {
		return
	}

	giver.Inventory[offerRes] -= amount
	giver.Inventory[wantRes] += amount
	receiver.Inventory[wantRes] -= amount
	receiver.Inventory[offerRes] += amount

	// Update trust
	giver.Trust[receiver.ID] += trustGain
	receiver.Trust[giver.ID] += trustGain

	// Improve trading skill
	giver.Skills["trading"] = math.Min(1.0, giver.Skills["trading"]+0.03*(1.0-giver.Skills["trading"]))
	receiver.Skills["trading"] = math.Min(1.0, receiver.Skills["trading"]+0.03*(1.0-receiver.Skills["trading"]))
}

// bestTradeResources finds what the agent has most of (to offer) and least of (to want).
func bestTradeResources(a *agent.Agent) (offer, want string) {
	maxStock := 0.0
	minStock := math.MaxFloat64

	for res, qty := range a.Inventory {
		// Skip crafted items and tools
		if res == "tools" || res == "medicine" {
			continue
		}
		if qty > maxStock {
			maxStock = qty
			offer = res
		}
		if qty < minStock {
			minStock = qty
			want = res
		}
	}

	// If agent only has one resource type, can't trade
	if offer == want {
		return "", ""
	}
	// Need meaningful surplus to trade
	if maxStock < tradeAmount {
		return "", ""
	}
	return offer, want
}

func abs(x int) int {
	if x < 0 {
		return -x
	}
	return x
}

func ScoreGoals(a *agent.Agent) GoalType {
	if a.Hunger > 30 {
		return GoalEat
	}
	if a.Hunger <= 20 && craftEngine != nil && craftEngine.CanCraft(a) != nil {
		return GoalCraft
	}
	// Trade when we have surplus of something
	if hasTradeSurplus(a) {
		return GoalTrade
	}
	return GoalExplore
}

func hasTradeSurplus(a *agent.Agent) bool {
	for res, qty := range a.Inventory {
		if res == "tools" || res == "medicine" {
			continue
		}
		if qty >= 4.0 { // enough surplus to consider trading
			return true
		}
	}
	return false
}
