WorldSim M5: Crafting and Skills -- Implementation Plan ​

For agentic workers: REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.

Goal: Add a config-driven recipe/crafting engine and agent skill progression. Agents can craft items from recipes defined in YAML. Skills improve with use (logarithmic curve) and affect gathering speed. Agents gain a GoalCraft goal when they have materials for useful recipes.

Architecture: New internal/craft package for the recipe engine. Add Skills map to Agent. Extend config types with RecipeDef. Update planner with GoalCraft. Gathering speed scales with gathering skill level.

Tech Stack: Pure Go, YAML configs, no new dependencies.

type recipesFile struct {
	Recipes []RecipeDef `yaml:"recipes"`
}

func LoadRecipes(path string) ([]RecipeDef, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("read recipes file: %w", err)
	}
	var f recipesFile
	if err := yaml.Unmarshal(data, &f); err != nil {
		return nil, fmt.Errorf("parse recipes file: %w", err)
	}
	return f.Recipes, nil
}

  • [ ] Step 2: Implement craft engine

Create internal/craft/engine.go:

go
package craft

import (
	"math"

	"github.com/alive/worldsim/internal/agent"
	"github.com/alive/worldsim/internal/config"
)

type Engine struct {
	recipes []config.RecipeDef
}

func NewEngine(recipes []config.RecipeDef) *Engine {
	return &Engine{recipes: recipes}
}

func (e *Engine) Recipes() []config.RecipeDef {
	return e.recipes
}

// CanCraft returns the first recipe the agent can craft, or nil.
func (e *Engine) CanCraft(a *agent.Agent) *config.RecipeDef {
	for i := range e.recipes {
		r := &e.recipes[i]
		if a.Skills["crafting"] < r.MinSkill {
			continue
		}
		canMake := true
		for _, input := range r.Inputs {
			if a.Inventory[input.Resource] < input.Quantity {
				canMake = false
				break
			}
		}
		if canMake {
			return r
		}
	}
	return nil
}

// Craft executes a recipe: consumes inputs, produces output, improves skill.
func (e *Engine) Craft(a *agent.Agent, recipe *config.RecipeDef) bool {
	// Verify materials
	for _, input := range recipe.Inputs {
		if a.Inventory[input.Resource] < input.Quantity {
			return false
		}
	}

	// Consume inputs
	for _, input := range recipe.Inputs {
		a.Inventory[input.Resource] -= input.Quantity
	}

	// Produce output
	a.Inventory[recipe.ID] += 1.0

	// Improve crafting skill (logarithmic curve)
	current := a.Skills["crafting"]
	gain := 0.05 * (1.0 - current) // diminishing returns
	a.Skills["crafting"] = math.Min(1.0, current+gain)

	return true
}

// GatheringSpeed returns the gathering speed multiplier based on tools in inventory.
func (e *Engine) GatheringSpeed(a *agent.Agent) float64 {
	speed := 1.0
	for _, r := range e.recipes {
		if r.Effects.GatheringSpeed > 0 && a.Inventory[r.ID] > 0 {
			if r.Effects.GatheringSpeed > speed {
				speed = r.Effects.GatheringSpeed
			}
		}
	}
	return speed
}

Update executeGather to use gathering speed and improve gathering skill:

go
func executeGather(ag *agent.Agent, w *world.World, resource string) bool {
	tile, err := w.GetTile(ag.X, ag.Y)
	if err != nil {
		return false
	}
	available := tile.Resources[resource]
	if available <= 0 {
		return false
	}
	// Gathering amount scales with skill and tools
	baseAmount := 1.0
	skillBonus := 1.0 + ag.Skills["gathering"]*0.5 // up to 1.5x from skill
	toolBonus := 1.0
	if craftEngine != nil {
		toolBonus = craftEngine.GatheringSpeed(ag)
	}
	amount := baseAmount * skillBonus * toolBonus
	if available < amount {
		amount = available
	}
	tile.Resources[resource] -= amount
	ag.Inventory[resource] += amount
	ag.Energy -= 0.5

	// Improve gathering skill
	current := ag.Skills["gathering"]
	gain := 0.02 * (1.0 - current)
	ag.Skills["gathering"] = math.Min(1.0, current+gain)
	return true
}

package ai

import "github.com/alive/worldsim/internal/agent"

type GoalType int

const (
	GoalEat GoalType = iota
	GoalExplore
	GoalCraft
)

func ScoreGoals(a *agent.Agent) GoalType {
	if a.Hunger > 30 {
		return GoalEat
	}
	// Craft if we can (checked by planner)
	if a.Hunger <= 20 && craftEngine != nil && craftEngine.CanCraft(a) != nil {
		return GoalCraft
	}
	return GoalExplore
}
